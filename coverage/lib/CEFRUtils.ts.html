export type CEFRLevel = 'A1' | 'A2' | 'B1' | 'B2' | 'C1' | 'C2';
 
export const getCEFRLevelName = (cefrLevel: CEFRLevel): string => {
  const names: Record<CEFRLevel, string> = {
    A1: 'Элементарный (A1)',
    A2: 'Базовый (A2)',
    B1: 'Средний (B1)',
    B2: 'Выше среднего (B2)',
    C1: 'Продвинутый (C1)',
    C2: 'В совершенстве (C2)'
  };
  return names[cefrLevel];
};

export const getCEFRScore = (percentage: number): { level: CEFRLevel; score: number } => {
  if (percentage >= 90) return { level: 'C2', score: percentage };
  if (percentage >= 80) return { level: 'C1', score: percentage };
  if (percentage >= 70) return { level: 'B2', score: percentage };
  if (percentage >= 60) return { level: 'B1', score: percentage };
  if (percentage >= 40) return { level: 'A2', score: percentage };
  return { level: 'A1', score: percentage };
};
